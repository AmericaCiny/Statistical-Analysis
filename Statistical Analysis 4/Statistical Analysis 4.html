stacked_Detergents <- stack(subset(Detergents, select = -Temperature))
names(stacked_Detergents) <- c("Score", "Detergent.Number")
bartlett.test(stacked_Detergents$Score ~ stacked_Detergents$Detergent.Number)
## 
##  Bartlett test of homogeneity of variances
## 
## data:  stacked_Detergents$Score by stacked_Detergents$Detergent.Number
## Bartlett's K-squared = 13.354, df = 4, p-value = 0.009668
Detergents_Gather <- Detergents %>%
  gather(key = "Detergent.Number", value = "Score", -Temperature)
Detergents_Gather$Temperature <- factor(Detergents_Gather$Temperature)

ANOVA_Ex4 <- aov(Detergents_Gather$Score ~ Detergents_Gather$Detergent.Number * Detergents_Gather$Temperature)
summary(ANOVA_Ex4)
